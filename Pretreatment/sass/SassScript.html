# SassScript

# 变量

# 普通变量

$width:sem;

#main {
    width: $width;
}

#main { width: sem; }

# 变量块级作用域

局部变量转换为全局变量可以添加 !global 声明

#nav {
    $height: 50px !global;
    height: $height;
}

#sidebar {
    height: $height;
}



#nav { height: 50px; }
#sidebar { height: 50px; }

# 数据类型

  • 数字，1, 2, 13, 10px
  • 字符串，有引号字符串与无引号字符串，"foo", 'bar', baz
  • 颜色，blue, #04a3f9, rgba(255,0,0,0.5)
  • 布尔型，true, false
  • 空值，null
  • 数组 (list)，用空格或逗号作分隔符，1.5em 1em 0 2em, Helvetica, Arial, sans-serif
  • maps, 相当于 JavaScript 的 object，(key1: value1, key2: value2)

# 运算

# 数字运算

p {
  width: 1in + 8pt;
}

p { width: 1.111in; }

p {
  font: 10px/8px;             //纯css属性，没有运算
  $width: 1000px;
  width: $width/2;            // 变量运算
  width: round(1.5)/2;        // 函数运算
  height: (500px/2);          // 括号运算
  margin-left: 5px + 8px/2px; // 混合运算
}

p {
  font: 10px/8px;
  width: 500px;
  height: 250px;
  margin-left: 9px; }

p {
  $font-size: 12px;
  $line-height: 30px;
  font: #{$font-size}/#{$line-height}; //变量运算
}

p { font: 12px/30px; }

# 颜色运算

# 十六进制

p {
  color: #010203 + #040506;
}

计算 01 + 04 = 05 02 + 05 = 07 03 + 06 = 09
p { color: #050709; }

p {
  color: #010203 * 2;
}
计算 01 * 2 = 02 02 * 2 = 04 03 * 2 = 06
p {
  color: #020406; }

# RGBA

p {
  color: rgba(255, 0, 0, 0.75) + rgba(0, 255, 0, 0.75);
}

p {
  color: rgba(255, 255, 0, 0.75); }

# alpha channel

$translucent-red: rgba(255, 0, 0, 0.5);
p {
  color: opacify($translucent-red, 0.3);
  background-color: transparentize($translucent-red, 0.25);
}

p {
  color: rgba(255, 0, 0, 0.8);
  background-color: rgba(255, 0, 0, 0.25); }

# 字符串运算

# + 可用于连接字符串

p {
  cursor: e + -resize;
}

p {
  cursor: e-resize; }

# 引号字符串

p:before {
  content: "Foo " + Bar;
  font-family: sans- + "serif";
}

p:before {
  content: "Foo Bar";
  font-family: sans-serif; }

# 运算表达式

p {
  margin: 3px + 4px auto;
}

p {
  margin: 7px auto; }

# #{}

p:before {
  content: "I ate #{5 + 10} pies!";
}

p:before {
  content: "I ate 15 pies!"; }

# null

$value: null;
p:before {
  content: "I ate #{$value} pies!";
}

p:before {
  content: "I ate pies!"; }

# 布尔运算

SassScript 支持布尔型的 and or 以及 not 运算

# 圆括号

p {
  width: 1em + (2em * 3);
}

p {
  width: 7em; }

# 函数

p {
  color: hsl(0, 100%, 50%);
}


p {
  color: #ff0000; }

# 插值语句

通过 #{} 插值语句可以在选择器或属性名中使用变量

$name: foo;
$attr: border;
p.#{$name} {
  #{$attr}-color: blue;
}

p.foo {
  border-color: blue; }

# 作用域插值

p {
  $font-size: 12px;
  $line-height: 30px;
  font: #{$font-size}/#{$line-height};
}

p {
  font: 12px/30px; }

# &标识符

@mixin does-parent-exist {
  @if & {
    &:hover {
      color: red;
    }
  } @else {
    a {
      color: red;
    }
  }
}

# 变量定义

变量定义 !default

$content: "First content";
$content: "Second content?" !default;
$new_content: "First time reference" !default;

#main {
  content: $content;
  new-content: $new_content;
}

#main {
  content: "First content";
  new-content: "First time reference"; }

变量是 null 空值时将视为未被 !default 赋值

$content: null;
$content: "Non-null content" !default;

#main {
  content: $content;
}

#main {
  content: "Non-null content"; }
